function generateCode(userRequest) { const codePanel = document.getElementById("code-panel"); const codeContent = document.getElementById("code-content"); const match = userRequest.match(/```([\s\S]*?)```/); if (match) { let code = match[1]; code = code.replace(/(function|const|let|var|return|if|else|for|while|class|import|from|export)/g, '$1'); code = code.replace(/("[^"]*"|'[^']*')/g, '$1'); code = code.replace(/(\/\/.*$)/gm, '$1'); codeContent.innerHTML = code; } else { codeContent.textContent = userRequest; } codePanel.classList.add("show"); } function closeCode() { document.getElementById("code-panel").classList.remove("show"); } function clearAppear() { const codePanel = document.getElementById("code-panel"); codePanel.style.opacity = "0"; setTimeout(() => { codePanel.style.opacity = "1"; }, 600); } function backToNormal() { const codePanel = document.getElementById("code-panel"); const codeContent = document.getElementById("code-content"); codeContent.textContent = ""; codePanel.classList.remove("show"); codePanel.style.opacity = "0"; } // Dark theme toggle with rainbow glowing drifting stars function toggleTheme() { document.body.classList.toggle("dark"); document.querySelectorAll(".star").forEach(s => s.remove()); if (document.body.classList.contains("dark")) { for (let i = 0; i < 25; i++) { const star = document.createElement("div"); star.className = "star"; star.style.top = Math.random() * 380 + "px"; star.style.left = Math.random() * 380 + "px"; const size = (Math.random() * 3 + 2); star.style.width = size + "px"; star.style.height = size + "px"; star.style.animationDelay = (Math.random() * 3) + "s"; document.getElementById("chat-box").appendChild(star); } } }
